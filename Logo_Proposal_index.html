    <script>

        function switchTab(tab) {

            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));

            document.querySelectorAll('nav button').forEach(b => b.classList.remove('nav-active'));

            

            document.getElementById('content-' + tab).classList.add('active');

            document.getElementById('tab-' + tab).classList.add('nav-active');

            

            if(tab === 'manual') initManualChart();

            if(tab === 'preview') initPreviewChart();

        }


        let mChart, pChart;


        function initManualChart() {

            if(mChart) mChart.destroy();

            const ctx = document.getElementById('manualChart').getContext('2d');

            mChart = new Chart(ctx, {

                type: 'radar',

                data: {

                    labels: ['Durabilidad', 'Honestidad', 'Precio', 'Diseño', 'Fe/Confianza'],

                    datasets: [{

                        label: 'Interés Nacional (%)',

                        data: [95, 90, 80, 85, 92],

                        backgroundColor: 'rgba(180, 83, 9, 0.2)',

                        borderColor: '#B45309',

                        pointBackgroundColor: '#B45309'

                    }]

                },

                options: { 

                    responsive: true, maintainAspectRatio: false,

                    scales: { r: { suggestedMin: 0, suggestedMax: 100, ticks: { display: false } } },

                    plugins: { legend: { display: false } }

                }

            });

        }
